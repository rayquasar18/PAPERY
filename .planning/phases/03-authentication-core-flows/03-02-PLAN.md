"""Auth request/response schemas — Pydantic v2 models for auth endpoints."""

import uuid as uuid_pkg
from datetime import datetime

from pydantic import BaseModel, ConfigDict, EmailStr, Field, field_validator


class RegisterRequest(BaseModel):
    """POST /auth/register input."""

    email: EmailStr
    password: str = Field(..., min_length=8, max_length=128)

    @field_validator("password")
    @classmethod
    def password_cannot_match_email(cls, v: str, info) -> str:
        email = info.data.get("email", "")
        if email and v.lower() == email.lower():
            raise ValueError("Password cannot match email address")
        return v


class LoginRequest(BaseModel):
    """POST /auth/login input."""

    email: EmailStr
    password: str


class UserPublicRead(BaseModel):
    """Public user data returned in auth responses. Never exposes id or hashed_password."""

    model_config = ConfigDict(from_attributes=True)

    uuid: uuid_pkg.UUID
    email: str
    display_name: str | None = None
    avatar_url: str | None = None
    is_verified: bool
    is_superuser: bool
    created_at: datetime


class AuthResponse(BaseModel):
    """Standard auth response body. Tokens are in cookies, NOT in this body."""

    user: UserPublicRead
    message: str


class MessageResponse(BaseModel):
    """Simple message response for endpoints that don't return user data."""

    message: str


class VerifyEmailRequest(BaseModel):
    """POST /auth/verify-email input."""

    token: str


class ResendVerificationRequest(BaseModel):
    """POST /auth/resend-verification input."""

    email: EmailStr


class TokenPayload(BaseModel):
    """Decoded JWT claims — internal only, never exposed in API."""

    sub: str  # user UUID as string
    jti: str  # unique token ID
    type: str  # "access" | "refresh"
    exp: int  # expiry timestamp
    iat: int  # issued-at timestamp
    family: str | None = None  # refresh token family ID (for replay detection)
    purpose: str | None = None  # "email_verification" for verification tokens
```

"""SMTP email sender utility.

Sends HTML emails via aiosmtplib (async) or falls back to logging
in local/test environments where SMTP is not configured.
"""

import logging
from email.message import EmailMessage

from app.configs import settings

logger = logging.getLogger(__name__)


async def send_email(to: str, subject: str, html_body: str) -> None:
    """Send an HTML email via SMTP.

    In local/test environments without SMTP configured, logs the email
    instead of sending (fail-safe for development).

    Args:
        to: Recipient email address
        subject: Email subject line
        html_body: HTML content of the email body
    """
    if not settings.SMTP_HOST:
        logger.warning(
            "SMTP not configured — email NOT sent. To: %s, Subject: %s, Body preview: %s",
            to,
            subject,
            html_body[:200],
        )
        return

    import aiosmtplib

    message = EmailMessage()
    message["From"] = settings.SMTP_FROM_EMAIL
    message["To"] = to
    message["Subject"] = subject
    message.set_content(html_body, subtype="html")

    try:
        await aiosmtplib.send(
            message,
            hostname=settings.SMTP_HOST,
            port=settings.SMTP_PORT,
            username=settings.SMTP_USER or None,
            password=settings.SMTP_PASSWORD or None,
            use_tls=settings.SMTP_PORT == 465,
            start_tls=settings.SMTP_PORT == 587,
        )
        logger.info("Email sent to %s: %s", to, subject)
    except Exception:
        logger.exception("Failed to send email to %s: %s", to, subject)
        # Do not raise — email failure should not block auth flows
        # The user can resend verification later
```

```python
"""Authentication service — business logic for user auth flows.

Handles registration, login, token management, email verification,
and refresh token rotation with replay detection.
"""

import logging
import uuid as uuid_pkg
from datetime import datetime, timedelta, timezone

from jose import JWTError, jwt
from passlib.context import CryptContext
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.configs import settings
from app.core.exceptions import (
    BadRequestError,
    ConflictError,
    NotFoundError,
    UnauthorizedError,
)
from app.infra.redis import client as redis_client
from app.models.user import User
from app.schemas.auth import TokenPayload, UserPublicRead
from app.utils.email import send_email

logger = logging.getLogger(__name__)

# Password hashing context — bcrypt with auto-upgrade
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

# Redis key prefixes
BLACKLIST_PREFIX = "blacklist:jti:"
FAMILY_PREFIX = "token_family:"


# ---------------------------------------------------------------------------
# Password utilities
# ---------------------------------------------------------------------------

def hash_password(plain_password: str) -> str:
    """Hash a password using bcrypt."""
    return pwd_context.hash(plain_password)


def verify_password(plain_password: str, hashed_password: str) -> bool:
    """Verify a password against its bcrypt hash (constant-time)."""
    return pwd_context.verify(plain_password, hashed_password)


# ---------------------------------------------------------------------------
# JWT token creation
# ---------------------------------------------------------------------------

def create_access_token(user_uuid: str) -> tuple[str, str]:
    """Create a signed JWT access token.

    Returns: (encoded_token, jti)
    """
    jti = str(uuid_pkg.uuid4())
    now = datetime.now(timezone.utc)
    payload = {
        "sub": user_uuid,
        "jti": jti,
        "type": "access",
        "iat": now,
        "exp": now + timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES),
    }
    token = jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
    return token, jti


def create_refresh_token(user_uuid: str, family_id: str | None = None) -> tuple[str, str, str]:
    """Create a signed JWT refresh token with family tracking.

    Returns: (encoded_token, jti, family_id)
    """
    jti = str(uuid_pkg.uuid4())
    if family_id is None:
        family_id = str(uuid_pkg.uuid4())
    now = datetime.now(timezone.utc)
    payload = {
        "sub": user_uuid,
        "jti": jti,
        "type": "refresh",
        "family": family_id,
        "iat": now,
        "exp": now + timedelta(days=settings.REFRESH_TOKEN_EXPIRE_DAYS),
    }
    token = jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
    return token, jti, family_id


def create_token_pair(user_uuid: str, family_id: str | None = None) -> tuple[str, str, str, str, str]:
    """Create both access and refresh tokens.

    Returns: (access_token, access_jti, refresh_token, refresh_jti, family_id)
    """
    access_token, access_jti = create_access_token(user_uuid)
    refresh_token, refresh_jti, fid = create_refresh_token(user_uuid, family_id)
    return access_token, access_jti, refresh_token, refresh_jti, fid


def decode_token(token: str) -> TokenPayload:
    """Decode and validate a JWT token. Raises UnauthorizedError on failure."""
    try:
        payload = jwt.decode(
            token,
            settings.SECRET_KEY,
            algorithms=[settings.ALGORITHM],
        )
        return TokenPayload(**payload)
    except JWTError as e:
        raise UnauthorizedError(
            detail="Invalid or expired token",
            error_code="TOKEN_INVALID",
        ) from e


# ---------------------------------------------------------------------------
# Token blacklist (Redis DB 0 — cache_client)
# ---------------------------------------------------------------------------

async def blacklist_token(jti: str, expire_seconds: int) -> None:
    """Add a JTI to the blacklist with TTL matching token expiry."""
    if redis_client.cache_client is None:
        raise RuntimeError("Redis cache client not initialized")
    await redis_client.cache_client.set(
        f"{BLACKLIST_PREFIX}{jti}", "1", ex=expire_seconds
    )


async def is_token_blacklisted(jti: str) -> bool:
    """Check if a JTI is in the blacklist."""
    if redis_client.cache_client is None:
        raise RuntimeError("Redis cache client not initialized")
    return await redis_client.cache_client.exists(f"{BLACKLIST_PREFIX}{jti}") > 0


# ---------------------------------------------------------------------------
# Token family tracking (for replay detection)
# ---------------------------------------------------------------------------

async def register_token_in_family(family_id: str, jti: str) -> None:
    """Register a refresh token JTI in its family set."""
    if redis_client.cache_client is None:
        raise RuntimeError("Redis cache client not initialized")
    family_key = f"{FAMILY_PREFIX}{family_id}"
    await redis_client.cache_client.sadd(family_key, jti)
    await redis_client.cache_client.expire(
        family_key, settings.REFRESH_TOKEN_EXPIRE_DAYS * 86400
    )


async def invalidate_token_family(family_id: str) -> None:
    """Invalidate ALL tokens in a family (replay attack response)."""
    if redis_client.cache_client is None:
        raise RuntimeError("Redis cache client not initialized")
    family_key = f"{FAMILY_PREFIX}{family_id}"
    family_jtis = await redis_client.cache_client.smembers(family_key)
    refresh_expire = settings.REFRESH_TOKEN_EXPIRE_DAYS * 86400
    for jti in family_jtis:
        await redis_client.cache_client.set(
            f"{BLACKLIST_PREFIX}{jti}", "1", ex=refresh_expire
        )
    await redis_client.cache_client.delete(family_key)
    logger.warning("Token family %s invalidated (replay detected) — %d tokens blacklisted", family_id, len(family_jtis))


# ---------------------------------------------------------------------------
# User operations
# ---------------------------------------------------------------------------

async def get_user_by_email(db: AsyncSession, email: str) -> User | None:
    """Fetch a non-deleted user by email."""
    stmt = select(User).where(User.email == email, User.deleted_at.is_(None))
    result = await db.execute(stmt)
    return result.scalar_one_or_none()


async def get_user_by_uuid(db: AsyncSession, user_uuid: str) -> User | None:
    """Fetch a non-deleted user by UUID."""
    stmt = select(User).where(
        User.uuid == user_uuid, User.deleted_at.is_(None)
    )
    result = await db.execute(stmt)
    return result.scalar_one_or_none()


async def register_user(db: AsyncSession, email: str, password: str) -> User:
    """Register a new user with email + hashed password.

    Raises ConflictError if email already exists.
    Sends verification email (non-blocking).
    """
    existing = await get_user_by_email(db, email)
    if existing is not None:
        raise ConflictError(
            detail="Email already registered",
            error_code="EMAIL_ALREADY_EXISTS",
        )

    user = User(
        email=email,
        hashed_password=hash_password(password),
        is_active=True,
        is_verified=False,
        is_superuser=False,
    )
    db.add(user)
    await db.commit()
    await db.refresh(user)

    # Send verification email (non-blocking — failure doesn't affect registration)
    await send_verification_email(email, str(user.uuid))

    return user


async def authenticate_user(db: AsyncSession, email: str, password: str) -> User:
    """Authenticate a user by email + password.

    Raises UnauthorizedError if credentials are invalid or account is inactive.
    """
    user = await get_user_by_email(db, email)
    if user is None:
        # Use constant-time comparison even on failure to prevent timing attacks
        pwd_context.dummy_verify()
        raise UnauthorizedError(
            detail="Invalid email or password",
            error_code="INVALID_CREDENTIALS",
        )
    if user.hashed_password is None:
        # OAuth-only user trying to login with password
        raise UnauthorizedError(
            detail="Invalid email or password",
            error_code="INVALID_CREDENTIALS",
        )
    if not verify_password(password, user.hashed_password):
        raise UnauthorizedError(
            detail="Invalid email or password",
            error_code="INVALID_CREDENTIALS",
        )
    if not user.is_active:
        raise UnauthorizedError(
            detail="Account is deactivated",
            error_code="ACCOUNT_INACTIVE",
        )
    return user


# ---------------------------------------------------------------------------
# Logout
# ---------------------------------------------------------------------------

async def logout_user(access_payload: TokenPayload, refresh_jti: str | None = None) -> None:
    """Blacklist current access token and optionally the refresh token."""
    # Calculate remaining TTL for access token
    now = int(datetime.now(timezone.utc).timestamp())
    access_remaining = max(access_payload.exp - now, 0)
    await blacklist_token(access_payload.jti, access_remaining)

    if refresh_jti:
        refresh_expire = settings.REFRESH_TOKEN_EXPIRE_DAYS * 86400
        await blacklist_token(refresh_jti, refresh_expire)


# ---------------------------------------------------------------------------
# Token refresh with rotation + replay detection
# ---------------------------------------------------------------------------

async def rotate_refresh_token(
    db: AsyncSession, old_payload: TokenPayload
) -> tuple[str, str, str, str, str]:
    """Rotate refresh token — issue new pair, blacklist old.

    If the old token is already blacklisted, this is a replay attack:
    invalidate the entire token family.

    Returns: (access_token, access_jti, refresh_token, refresh_jti, family_id)
    """
    # Check for replay attack
    if await is_token_blacklisted(old_payload.jti):
        # REPLAY DETECTED — invalidate entire family
        if old_payload.family:
            await invalidate_token_family(old_payload.family)
        raise UnauthorizedError(
            detail="Token has been revoked — possible replay attack. All sessions terminated.",
            error_code="TOKEN_FAMILY_REVOKED",
        )

    # Verify user still exists and is active
    user = await get_user_by_uuid(db, old_payload.sub)
    if user is None or not user.is_active:
        raise UnauthorizedError(
            detail="User not found or inactive",
            error_code="INVALID_CREDENTIALS",
        )

    # Blacklist the old refresh token
    now = int(datetime.now(timezone.utc).timestamp())
    old_remaining = max(old_payload.exp - now, 0)
    await blacklist_token(old_payload.jti, old_remaining)

    # Issue new token pair with SAME family ID
    family_id = old_payload.family or str(uuid_pkg.uuid4())
    access_token, access_jti, refresh_token, refresh_jti, fid = create_token_pair(
        str(user.uuid), family_id
    )

    # Register new refresh JTI in the family
    await register_token_in_family(fid, refresh_jti)

    return access_token, access_jti, refresh_token, refresh_jti, fid


# ---------------------------------------------------------------------------
# Email verification
# ---------------------------------------------------------------------------

def create_email_verification_token(user_uuid: str) -> str:
    """Create a signed JWT for email verification (24h expiry)."""
    now = datetime.now(timezone.utc)
    payload = {
        "sub": user_uuid,
        "purpose": "email_verification",
        "jti": str(uuid_pkg.uuid4()),
        "iat": now,
        "exp": now + timedelta(hours=24),
    }
    return jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.ALGORITHM)


async def verify_email(db: AsyncSession, token: str) -> User:
    """Verify a user's email using the verification token.

    Decodes the JWT, checks purpose, sets is_verified=True.
    """
    try:
        payload = jwt.decode(
            token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM]
        )
    except JWTError as e:
        raise UnauthorizedError(
            detail="Invalid or expired verification token",
            error_code="VERIFICATION_TOKEN_INVALID",
        ) from e

    if payload.get("purpose") != "email_verification":
        raise BadRequestError(
            detail="Invalid token purpose",
            error_code="INVALID_TOKEN_PURPOSE",
        )

    user_uuid = payload.get("sub")
    if not user_uuid:
        raise BadRequestError(
            detail="Invalid verification token",
            error_code="VERIFICATION_TOKEN_INVALID",
        )

    user = await get_user_by_uuid(db, user_uuid)
    if user is None:
        raise NotFoundError(
            detail="User not found",
            error_code="USER_NOT_FOUND",
        )

    if user.is_verified:
        return user  # Already verified — idempotent

    user.is_verified = True
    await db.commit()
    await db.refresh(user)
    return user


async def send_verification_email(email: str, user_uuid: str) -> None:
    """Build and send the email verification message.

    Public API — called from register_user() internally and from
    the resend-verification route handler externally.

    Uses settings.FRONTEND_URL to construct the verification link.
    """
    token = create_email_verification_token(user_uuid)
    verification_url = f"{settings.FRONTEND_URL}/verify-email?token={token}"
    html_body = f"""
    <h2>Verify your email</h2>
    <p>Welcome to {settings.APP_NAME}! Please verify your email address by clicking the link below:</p>
    <p><a href="{verification_url}">Verify Email</a></p>
    <p>This link expires in 24 hours.</p>
    <p>If you did not create an account, please ignore this email.</p>
    """
    await send_email(to=email, subject=f"{settings.APP_NAME} — Verify your email", html_body=html_body)


# ---------------------------------------------------------------------------
# Superuser bootstrap
# ---------------------------------------------------------------------------

async def create_first_superuser(db: AsyncSession) -> User | None:
    """Create the bootstrap superuser from AdminConfig settings.

    Returns the user if created, None if already exists.
    """
    existing = await get_user_by_email(db, settings.ADMIN_EMAIL)
    if existing is not None:
        logger.info("Superuser %s already exists, skipping creation", settings.ADMIN_EMAIL)
        return None

    user = User(
        email=settings.ADMIN_EMAIL,
        hashed_password=hash_password(settings.ADMIN_PASSWORD),
        is_active=True,
        is_verified=True,
        is_superuser=True,
        display_name="Admin",
    )
    db.add(user)
    await db.commit()
    await db.refresh(user)
    logger.info("Superuser created: %s", settings.ADMIN_EMAIL)
    return user
```

**Key implementation details:**
- `pwd_context.dummy_verify()` on user-not-found prevents timing attacks (attacker can't distinguish "user doesn't exist" from "wrong password" by response time)
- Token blacklist uses remaining TTL (not fixed TTL) — so blacklist entries auto-expire with the token
- Family invalidation loops through all family JTIs and blacklists each one
- `send_verification_email` is a **public function** (no underscore prefix) — called both internally by `register_user()` and externally by the resend-verification route handler
- `send_verification_email` uses `settings.FRONTEND_URL` (from `AppConfig`, default `http://localhost:3000`) to construct valid verification URLs
- `create_first_superuser` is idempotent — safe to call on every startup

## Threat Model

| Threat | Severity | Mitigation |
|--------|----------|------------|
| Timing attack on login (user enumeration) | High | `pwd_context.dummy_verify()` called when user not found — constant-time response regardless of user existence |
| Password brute force | High | Rate limiting in Plan 03-03; bcrypt adaptive hashing with high work factor |
| Credential stuffing | High | Rate limiting in Plan 03-03 (5 attempts/min per IP) |
| Refresh token replay | Critical | Token family concept — replay of blacklisted refresh token invalidates ALL tokens in family |
| Token theft via XSS | Critical | Tokens in HttpOnly cookies (Plan 03-03) — not accessible to JavaScript |
| JWT secret compromise | Critical | SECRET_KEY validated at startup (Phase 1); rotation invalidates all tokens by design (D-01) |
| Email verification bypass | Medium | Signed JWT with `purpose` claim — cannot reuse access tokens as verification tokens |
| Password stored reversibly | Critical | bcrypt via passlib — one-way adaptive hash with salt |
| Weak password | Medium | Pydantic validator: 8+ chars, cannot match email (NIST SP 800-63B aligned) |
