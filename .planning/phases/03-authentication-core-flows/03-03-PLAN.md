```python
"""Rate limiting utility using Redis sliding window counter.

Uses Redis DB 2 (rate_limit_client) for rate limit tracking.
"""

import logging

from app.core.exceptions import RateLimitedError
from app.infra.redis import client as redis_client

logger = logging.getLogger(__name__)


async def check_rate_limit(
    key: str,
    max_requests: int,
    window_seconds: int,
) -> None:
    """Check and enforce rate limit using Redis INCR + EXPIRE.

    Args:
        key: Rate limit key (e.g., "rate_limit:login:127.0.0.1")
        max_requests: Maximum allowed requests in the window
        window_seconds: Time window in seconds

    Raises:
        RateLimitedError: If rate limit exceeded (429 with Retry-After header)
        RuntimeError: If Redis rate_limit_client is not initialized
    """
    if redis_client.rate_limit_client is None:
        raise RuntimeError("Redis rate_limit client not initialized")

    count = await redis_client.rate_limit_client.incr(key)
    if count == 1:
        await redis_client.rate_limit_client.expire(key, window_seconds)

    if count > max_requests:
        ttl = await redis_client.rate_limit_client.ttl(key)
        if ttl < 0:
            ttl = window_seconds
        raise RateLimitedError(
            detail=f"Too many requests. Retry after {ttl} seconds.",
            error_code="RATE_LIMITED",
            headers={"Retry-After": str(ttl)},
        )
```

Key design decisions:
- Uses `INCR` + `EXPIRE` pattern (simple, atomic per operation)
- Uses `rate_limit_client` (Redis DB 2), NOT `cache_client` (DB 0)
- Returns 429 with `Retry-After` header per HTTP spec
- TTL fallback to `window_seconds` if Redis TTL returns -1 (edge case)

```python
"""Shared FastAPI dependencies — auth, database session.

These dependencies are used by route handlers via Depends().
"""

import logging

from fastapi import Depends, Request
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.db.session import get_session
from app.core.exceptions import ForbiddenError, UnauthorizedError
from app.models.user import User
from app.services import auth_service

logger = logging.getLogger(__name__)


async def get_current_user(
    request: Request,
    db: AsyncSession = Depends(get_session),
) -> User:
    """Extract and validate access token from HttpOnly cookie.

    Flow:
    1. Read access_token cookie from request
    2. Decode JWT → validate signature + expiry
    3. Check token blacklist (Redis)
    4. Load User from database by UUID
    5. Return User ORM instance

    Raises UnauthorizedError (401) if any step fails.
    """
    token = request.cookies.get("access_token")
    if not token:
        raise UnauthorizedError(
            detail="Authentication required",
            error_code="TOKEN_MISSING",
        )

    payload = auth_service.decode_token(token)

    if payload.type != "access":
        raise UnauthorizedError(
            detail="Invalid token type",
            error_code="TOKEN_INVALID",
        )

    if await auth_service.is_token_blacklisted(payload.jti):
        raise UnauthorizedError(
            detail="Token has been revoked",
            error_code="TOKEN_REVOKED",
        )

    user = await auth_service.get_user_by_uuid(db, payload.sub)
    if user is None:
        raise UnauthorizedError(
            detail="User not found",
            error_code="USER_NOT_FOUND",
        )

    return user


async def get_current_active_user(
    user: User = Depends(get_current_user),
) -> User:
    """Ensure the current user is active.

    Wraps get_current_user and additionally checks is_active flag.
    Raises ForbiddenError (403) if user is deactivated.
    """
    if not user.is_active:
        raise ForbiddenError(
            detail="Account is deactivated",
            error_code="ACCOUNT_INACTIVE",
        )
    return user


async def get_current_superuser(
    user: User = Depends(get_current_active_user),
) -> User:
    """Ensure the current user is an active superuser.

    Wraps get_current_active_user and additionally checks is_superuser flag.
    Raises ForbiddenError (403) if user is not a superuser.
    """
    if not user.is_superuser:
        raise ForbiddenError(
            detail="Superuser access required",
            error_code="SUPERUSER_REQUIRED",
        )
    return user
```

Dependency chain: `get_current_superuser` → `get_current_active_user` → `get_current_user` → `get_session`

```python
"""Auth API routes — registration, login, logout, refresh, verification.

All tokens are transported via HttpOnly cookies. Response bodies contain
user data and messages, NEVER tokens.
"""

import logging

from fastapi import APIRouter, Depends, Request, Response
from sqlalchemy.ext.asyncio import AsyncSession

from app.api.dependencies import get_current_user
from app.configs import settings
from app.core.db.session import get_session
from app.core.exceptions import UnauthorizedError
from app.models.user import User
from app.schemas.auth import (
    AuthResponse,
    LoginRequest,
    MessageResponse,
    RegisterRequest,
    ResendVerificationRequest,
    UserPublicRead,
    VerifyEmailRequest,
)
from app.services import auth_service
from app.utils.rate_limit import check_rate_limit

logger = logging.getLogger(__name__)

router = APIRouter(prefix="/auth", tags=["auth"])


def _set_auth_cookies(response: Response, access_token: str, refresh_token: str) -> None:
    """Set access and refresh token cookies on the response.

    Access token cookie:
    - Path: / (available to all endpoints)
    - Max-Age: ACCESS_TOKEN_EXPIRE_MINUTES * 60

    Refresh token cookie:
    - Path: /api/v1/auth/refresh (scoped to refresh endpoint only)
    - Max-Age: REFRESH_TOKEN_EXPIRE_DAYS * 86400
    """
    is_secure = settings.ENVIRONMENT != "local"

    response.set_cookie(
        key="access_token",
        value=access_token,
        httponly=True,
        secure=is_secure,
        samesite="lax",
        max_age=settings.ACCESS_TOKEN_EXPIRE_MINUTES * 60,
        path="/",
    )
    response.set_cookie(
        key="refresh_token",
        value=refresh_token,
        httponly=True,
        secure=is_secure,
        samesite="lax",
        max_age=settings.REFRESH_TOKEN_EXPIRE_DAYS * 86400,
        path="/api/v1/auth/refresh",
    )


def _clear_auth_cookies(response: Response) -> None:
    """Delete auth cookies from the response."""
    response.delete_cookie(key="access_token", path="/")
    response.delete_cookie(key="refresh_token", path="/api/v1/auth/refresh")


@router.post("/register", response_model=AuthResponse, status_code=201)
async def register(
    body: RegisterRequest,
    request: Request,
    response: Response,
    db: AsyncSession = Depends(get_session),
) -> AuthResponse:
    """Register a new user with email + password.

    Rate limit: 3 requests per minute per IP.
    Sets JWT cookies on success.
    """
    client_ip = request.client.host if request.client else "unknown"
    await check_rate_limit(f"rate_limit:register:{client_ip}", max_requests=3, window_seconds=60)

    user = await auth_service.register_user(db, body.email, body.password)

    # Issue tokens immediately (user can login before verification — D-11)
    access_token, access_jti, refresh_token, refresh_jti, family_id = auth_service.create_token_pair(str(user.uuid))
    await auth_service.register_token_in_family(family_id, refresh_jti)

    _set_auth_cookies(response, access_token, refresh_token)

    return AuthResponse(
        user=UserPublicRead.model_validate(user),
        message="Registration successful. Please check your email to verify your account.",
    )


@router.post("/login", response_model=AuthResponse)
async def login(
    body: LoginRequest,
    request: Request,
    response: Response,
    db: AsyncSession = Depends(get_session),
) -> AuthResponse:
    """Log in with email + password.

    Rate limit: 5 requests per minute per IP.
    Sets JWT cookies on success.
    """
    client_ip = request.client.host if request.client else "unknown"
    await check_rate_limit(f"rate_limit:login:{client_ip}", max_requests=5, window_seconds=60)

    user = await auth_service.authenticate_user(db, body.email, body.password)

    access_token, access_jti, refresh_token, refresh_jti, family_id = auth_service.create_token_pair(str(user.uuid))
    await auth_service.register_token_in_family(family_id, refresh_jti)

    _set_auth_cookies(response, access_token, refresh_token)

    return AuthResponse(
        user=UserPublicRead.model_validate(user),
        message="Login successful",
    )


@router.post("/logout", response_model=MessageResponse)
async def logout(
    request: Request,
    response: Response,
    user: User = Depends(get_current_user),
) -> MessageResponse:
    """Log out — blacklist tokens and clear cookies.

    Requires valid access token cookie.
    """
    access_token = request.cookies.get("access_token")
    refresh_token = request.cookies.get("refresh_token")

    if access_token:
        access_payload = auth_service.decode_token(access_token)
        refresh_jti = None
        if refresh_token:
            try:
                refresh_payload = auth_service.decode_token(refresh_token)
                refresh_jti = refresh_payload.jti
            except Exception:
                pass  # Refresh token may already be expired — that's OK
        await auth_service.logout_user(access_payload, refresh_jti)

    _clear_auth_cookies(response)

    return MessageResponse(message="Logged out successfully")


@router.post("/refresh", response_model=AuthResponse)
async def refresh(
    request: Request,
    response: Response,
    db: AsyncSession = Depends(get_session),
) -> AuthResponse:
    """Refresh access token using refresh token cookie.

    Implements refresh token rotation:
    - Old refresh token is blacklisted
    - New access + refresh token pair is issued
    - If old refresh token is already blacklisted (replay), entire family is invalidated
    """
    refresh_token_value = request.cookies.get("refresh_token")
    if not refresh_token_value:
        raise UnauthorizedError(
            detail="Refresh token required",
            error_code="TOKEN_MISSING",
        )

    old_payload = auth_service.decode_token(refresh_token_value)

    if old_payload.type != "refresh":
        raise UnauthorizedError(
            detail="Invalid token type",
            error_code="TOKEN_INVALID",
        )

    access_token, access_jti, new_refresh_token, refresh_jti, family_id = (
        await auth_service.rotate_refresh_token(db, old_payload)
    )

    _set_auth_cookies(response, access_token, new_refresh_token)

    # Load user for response body
    user = await auth_service.get_user_by_uuid(db, old_payload.sub)
    if user is None:
        raise UnauthorizedError(detail="User not found", error_code="USER_NOT_FOUND")

    return AuthResponse(
        user=UserPublicRead.model_validate(user),
        message="Token refreshed",
    )


@router.get("/me", response_model=UserPublicRead)
async def get_me(
    user: User = Depends(get_current_user),
) -> UserPublicRead:
    """Get current user profile.

    Requires valid access token cookie.
    """
    return UserPublicRead.model_validate(user)


@router.post("/verify-email", response_model=MessageResponse)
async def verify_email(
    body: VerifyEmailRequest,
    db: AsyncSession = Depends(get_session),
) -> MessageResponse:
    """Verify email address using the token from verification email."""
    await auth_service.verify_email(db, body.token)
    return MessageResponse(message="Email verified successfully")


@router.post("/resend-verification", response_model=MessageResponse)
async def resend_verification(
    body: ResendVerificationRequest,
    request: Request,
    db: AsyncSession = Depends(get_session),
) -> MessageResponse:
    """Resend email verification link.

    Rate limit: 1 request per 60 seconds per email.
    Returns success message even if email is not found (prevents enumeration).
    """
    await check_rate_limit(
        f"rate_limit:resend_verification:{body.email}",
        max_requests=1,
        window_seconds=60,
    )

    user = await auth_service.get_user_by_email(db, body.email)
    if user is not None and not user.is_verified:
        await auth_service.send_verification_email(user.email, str(user.uuid))

    # Always return success to prevent email enumeration
    return MessageResponse(
        message="If the email is registered and unverified, a verification link has been sent."
    )
```

**Critical security decisions:**
- `_set_auth_cookies`: `httponly=True`, `secure=True` in non-local, `samesite="lax"`, refresh cookie scoped to `/api/v1/auth/refresh`

- Register endpoint returns 201, not 200
- `resend_verification` returns same message regardless of email existence (prevents enumeration)
- Rate limits: register 3/min, login 5/min, resend-verification 1/60s per email

## Threat Model

| Threat | Severity | Mitigation |
|--------|----------|------------|
| XSS token theft | Critical | HttpOnly=True on both cookies — JavaScript cannot read tokens |
| CSRF attacks | High | SameSite=Lax cookies — cross-origin POST requests don't send cookies |
| Session fixation | High | New token pair issued on every login + rotation on refresh |
| Token replay (refresh) | Critical | Token family tracking — replay blacklists entire family |
| Credential stuffing | High | Rate limit: login 5/min per IP |
| Registration spam | Medium | Rate limit: register 3/min per IP |
| Email enumeration via register | Medium | ConflictError on duplicate email (trade-off: user feedback vs. enumeration) |
| Email enumeration via resend-verification | Medium | Always returns same success message regardless of email existence |
| Refresh cookie scope | Medium | Refresh cookie path set to `/api/v1/auth/refresh` only — not sent on other requests |
| Cookie downgrade attack | High | Secure=True in non-local environments — cookies only sent over HTTPS |
