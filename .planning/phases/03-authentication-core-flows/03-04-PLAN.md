class TestPasswordHashing:
    """Tests for bcrypt password hashing."""

    def test_hash_password_returns_bcrypt_hash(self):
        hashed = hash_password("testpassword")
        assert hashed.startswith("$2b$")
        assert len(hashed) == 60

    def test_verify_correct_password(self):
        hashed = hash_password("testpassword")
        assert verify_password("testpassword", hashed) is True

    def test_verify_wrong_password(self):
        hashed = hash_password("testpassword")
        assert verify_password("wrongpassword", hashed) is False

    def test_different_passwords_different_hashes(self):
        hash1 = hash_password("password1")
        hash2 = hash_password("password2")
        assert hash1 != hash2

    def test_same_password_different_hashes(self):
        """bcrypt salts ensure same input produces different hashes."""
        hash1 = hash_password("samepassword")
        hash2 = hash_password("samepassword")
        assert hash1 != hash2  # Different salts


class TestJWTTokens:
    """Tests for JWT token creation and decoding."""

    def test_create_access_token(self):
        user_uuid = str(uuid_pkg.uuid4())
        token, jti = create_access_token(user_uuid)
        assert isinstance(token, str)
        assert len(token) > 0
        assert isinstance(jti, str)

    def test_decode_access_token(self):
        user_uuid = str(uuid_pkg.uuid4())
        token, jti = create_access_token(user_uuid)
        payload = decode_token(token)
        assert payload.sub == user_uuid
        assert payload.jti == jti
        assert payload.type == "access"
        assert payload.family is None

    def test_create_refresh_token(self):
        user_uuid = str(uuid_pkg.uuid4())
        token, jti, family_id = create_refresh_token(user_uuid)
        assert isinstance(token, str)
        assert isinstance(family_id, str)

    def test_decode_refresh_token(self):
        user_uuid = str(uuid_pkg.uuid4())
        token, jti, family_id = create_refresh_token(user_uuid)
        payload = decode_token(token)
        assert payload.sub == user_uuid
        assert payload.type == "refresh"
        assert payload.family == family_id

    def test_refresh_token_with_existing_family(self):
        user_uuid = str(uuid_pkg.uuid4())
        family_id = str(uuid_pkg.uuid4())
        token, jti, fid = create_refresh_token(user_uuid, family_id)
        payload = decode_token(token)
        assert payload.family == family_id
        assert fid == family_id

    def test_create_token_pair(self):
        user_uuid = str(uuid_pkg.uuid4())
        access_token, access_jti, refresh_token, refresh_jti, family_id = create_token_pair(user_uuid)
        assert access_jti != refresh_jti  # Different JTIs
        access_payload = decode_token(access_token)
        refresh_payload = decode_token(refresh_token)
        assert access_payload.type == "access"
        assert refresh_payload.type == "refresh"
        assert access_payload.sub == refresh_payload.sub == user_uuid

    def test_decode_invalid_token(self):
        from app.core.exceptions import UnauthorizedError
        with pytest.raises(UnauthorizedError, match="Invalid or expired token"):
            decode_token("not.a.valid.jwt.token")

    def test_decode_expired_token(self):
        """Tokens with past exp should raise UnauthorizedError."""
        from jose import jwt
        from app.configs import settings
        from app.core.exceptions import UnauthorizedError

        payload = {
            "sub": str(uuid_pkg.uuid4()),
            "jti": str(uuid_pkg.uuid4()),
            "type": "access",
            "iat": datetime.now(timezone.utc) - timedelta(hours=2),
            "exp": datetime.now(timezone.utc) - timedelta(hours=1),
        }
        expired_token = jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
        with pytest.raises(UnauthorizedError):
            decode_token(expired_token)

import asyncio
import logging
import sys

logger = logging.getLogger(__name__)


async def main() -> None:
    """Initialize database and create the first superuser."""
    from app.configs import settings
    from app.core.db import session as db_session
