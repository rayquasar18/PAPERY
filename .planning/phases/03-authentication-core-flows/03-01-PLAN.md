```python
"""User and OAuthAccount ORM models."""

from sqlalchemy import Boolean, String, Text
from sqlalchemy.orm import Mapped, mapped_column, relationship

from app.models.base import Base, SoftDeleteMixin, TimestampMixin, UUIDMixin


class User(Base, UUIDMixin, TimestampMixin, SoftDeleteMixin):
    """User account — primary identity entity.

    Fields:
    - email: unique login identifier (indexed)
    - hashed_password: bcrypt hash (nullable for OAuth-only users)
    - display_name: optional public name
    - avatar_url: optional profile image URL
    - is_active: can this user log in? (default True)
    - is_verified: has email been verified? (default False)
    - is_superuser: admin privileges? (default False)
    """

    __tablename__ = "user"

    email: Mapped[str] = mapped_column(
        String(320), unique=True, index=True, nullable=False
    )
    hashed_password: Mapped[str | None] = mapped_column(
        Text, nullable=True, default=None
    )
    display_name: Mapped[str | None] = mapped_column(
        String(100), nullable=True, default=None
    )
    avatar_url: Mapped[str | None] = mapped_column(
        Text, nullable=True, default=None
    )
    is_active: Mapped[bool] = mapped_column(
        Boolean, default=True, server_default="true", nullable=False
    )
    is_verified: Mapped[bool] = mapped_column(
        Boolean, default=False, server_default="false", nullable=False
    )
    is_superuser: Mapped[bool] = mapped_column(
        Boolean, default=False, server_default="false", nullable=False
    )

    # Relationship to OAuth accounts (one user → many OAuth providers)
    oauth_accounts: Mapped[list["OAuthAccount"]] = relationship(
        "OAuthAccount", back_populates="user", cascade="all, delete-orphan"
    )

    def __repr__(self) -> str:
        return f"<User uuid={self.uuid} email={self.email}>"
```

Key points:
- `email` uses String(320) — max email length per RFC 5321
- `hashed_password` is nullable (for OAuth-only users)
- Boolean fields use `server_default` for database-level defaults
- Inherits `id` (BigInteger PK) from `Base`, `uuid` from `UUIDMixin`, `created_at`/`updated_at` from `TimestampMixin`, `deleted_at` from `SoftDeleteMixin`

```python
from sqlalchemy import BigInteger, ForeignKey, String, UniqueConstraint

class OAuthAccount(Base, TimestampMixin):
    """OAuth provider link — connects a User to an external identity provider.

    One User can have multiple OAuth accounts (Google, GitHub, etc.).
    Phase 3 creates the table; Phase 4 implements the full OAuth flow.
    """

    __tablename__ = "oauth_account"
    __table_args__ = (
        UniqueConstraint("provider", "provider_user_id", name="uq_oauth_provider_user"),
    )

    user_id: Mapped[int] = mapped_column(
        BigInteger, ForeignKey("user.id", ondelete="CASCADE"), nullable=False, index=True
    )
    provider: Mapped[str] = mapped_column(
        String(50), nullable=False  # "google", "github"
    )
    provider_user_id: Mapped[str] = mapped_column(
        String(255), nullable=False  # external provider's user ID
    )
    provider_email: Mapped[str | None] = mapped_column(
        String(320), nullable=True  # email from the provider
    )

    # Back-reference to User
    user: Mapped["User"] = relationship("User", back_populates="oauth_accounts")

    def __repr__(self) -> str:
        return f"<OAuthAccount provider={self.provider} user_id={self.user_id}>"
```

Key points:
- Does NOT use UUIDMixin or SoftDeleteMixin (internal join table, not API-facing)
- Has `TimestampMixin` for audit trail
- Unique constraint on `(provider, provider_user_id)` — one provider account links to one User
- ForeignKey with `ondelete="CASCADE"` — if User is hard-deleted, OAuth links cascade

## Threat Model

| Threat | Severity | Mitigation |
|--------|----------|------------|
| Email enumeration via unique constraint error | Medium | Handled in Plan 03-03 (generic error messages on register/login) |
| Password stored in plaintext | Critical | `hashed_password` column — actual hashing in Plan 03-02 (passlib bcrypt) |
| OAuth account hijacking (linking wrong user) | High | UniqueConstraint on (provider, provider_user_id) prevents duplicate linking |
| User ID enumeration via API | High | UUID used as public identifier (UUIDMixin); int `id` never exposed in API |
| Soft-deleted user data exposure | Medium | SoftDeleteMixin + query filters in service layer (Plan 03-02/03-03) |
