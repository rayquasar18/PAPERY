```typescript
import { z } from 'zod';

export const loginSchema = z.object({
  email: z.email({ error: 'Please enter a valid email address.' }),
  password: z.string().min(8, { error: 'Password must be at least 8 characters.' }),
});

export const registerSchema = z.object({
  displayName: z.string().min(2, { error: 'Name must be between 2 and 50 characters.' }).max(50, { error: 'Name must be between 2 and 50 characters.' }),
  email: z.email({ error: 'Please enter a valid email address.' }),
  password: z.string().min(8, { error: 'Password must be at least 8 characters with uppercase, lowercase, and number.' }),
  confirmPassword: z.string(),
}).refine((data) => data.password === data.confirmPassword, {
  error: 'Passwords do not match.',
  path: ['confirmPassword'],
});

export const forgotPasswordSchema = z.object({
  email: z.email({ error: 'Please enter a valid email address.' }),
});

export const resetPasswordSchema = z.object({
  password: z.string().min(8, { error: 'Password must be at least 8 characters.' }),
  confirmPassword: z.string(),
}).refine((data) => data.password === data.confirmPassword, {
  error: 'Passwords do not match.',
  path: ['confirmPassword'],
});

export type LoginInput = z.infer<typeof loginSchema>;
export type RegisterInput = z.infer<typeof registerSchema>;
export type ForgotPasswordInput = z.infer<typeof forgotPasswordSchema>;
export type ResetPasswordInput = z.infer<typeof resetPasswordSchema>;
```

## Threat Model

<threat_model>
| Threat | Severity | Likelihood | Mitigation |
|--------|----------|------------|------------|
| Token exposure via JavaScript | CRITICAL | NONE | Backend uses HttpOnly cookies; frontend NEVER handles JWT in JS |
| CSRF via cross-origin request | MEDIUM | LOW | SameSite=Lax cookies; CORS explicit origin; withCredentials requires exact origin |
| XSS via API error messages | MEDIUM | LOW | Error normalization sanitizes backend response; React auto-escapes JSX |
| Refresh token race condition | LOW | LOW | Queue pattern with isRefreshing flag serializes refresh calls |
</threat_model>
